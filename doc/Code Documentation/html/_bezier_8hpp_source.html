25 // Stores information about all the (D+1) Berstein polynomials of degree D (denoted Bi(t) for i=0 to D),
26 // so you can use this object to efficiently compute their values and derivatives at arbitrary values of t.
27 // NOTE: The coefficient in front of each polynomial is 1, we're not using the Pascal triangle thing in this object.
28 class BernsteinPolynomial
29 {
30 
31 public:
32  BernsteinPolynomial( unsigned int degree );
33  ~BernsteinPolynomial();
34  // Returns the degree of the polynomial
35  unsigned int getD() const;
36  // Returns a matrix of values and derivatives of each polynomial.
37  // First index i = which polynomial. From 0 to D
38  // Second index j = how many times differentiated. From 0 to D
39  MatrixMxN compute( double t ) const;
40  // For this one, it only returns the first column of the above matrix.
41  MatrixMxN computeFOnly( double t ) const;
42 
43 private:
44  // highest power of t in the expression of B(t)
45  unsigned int mD;
46  // This is like a 3-d array. The coefficients are all integers.
47  // First index i = which of the mDegree polynomials. From 0 to D
48  // Second index j = power of t. From 0 to D-i
49  // Third index k = power of (1-t). From 0 to i
50  // The number of times differentiated is implied by the three indices.
51  // The indices used in the implementation are are i, j*(i+1)+k
52  int** mB;
53 
54 };

56 // This is a function f(x,y) for x,y from 0 to 1.
57 // The matrix K has all the coefficients.
58 // f(x,y) = sum(over i and j)( Bi(x)*Bj(y)*Kij )
59 // where B are degree (2n-1) Bernstein Polynomials
60 class BezierPatch
61 {
62 public:
63  // K will be all zero in this constructor.
64  BezierPatch( const BernsteinPolynomial* BP );
65  // The argument matrices have to be in accordance with the specified n.
66  BezierPatch( const BernsteinPolynomial* BP, MatrixMxN K );
67  // Specify the value and some first, second etc derivatives at each corner.
68  // Each matrix has the same format as the F returned in compute(x,y).
69  // BP must have an odd degree so K can be a 2n*2n matrix, where 2n=D+1.
70  // BTildaInv0 is the relevant quarter of BTilda0^-1. Same for BTildaInv1.
71  BezierPatch( const BernsteinPolynomial* BP,
72  const MatrixMxN& BTildaInv0,
73  const MatrixMxN& BTildaInv1,
74  const MatrixMxN& f00, const MatrixMxN& f01,
75  const MatrixMxN& f10, const MatrixMxN& f11 );
76  ~BezierPatch();
77 
78  // Returns mK (see below)
79  const MatrixMxN& getK() const;
80  // Returns the Berstein polynomial object that this uses.
81  const BernsteinPolynomial* getBP() const;
82  // x and y should both be between 0 and 1.
83  // Returns the value and some derivatives of f at (x,y).
84  // For example, when n = 2, it returns
85  // [ f, fy, fyy, fyyy;
86  // fx, fxy, fxyy, fxyyy;
87  // fxx, fxxy, fxxyy, fxxyyy;
88  // fxxx, fxxxy, fxxxyy, fxxxyyy ]
89  MatrixMxN compute( double x, double y ) const;
90  // Returns the topleft value in the above matrix.
91  // computeFOnly(x,y) is more efficient than compute(x,y).
92  double computeFOnly( double x, double y ) const;
93 
94 private:
95  const BernsteinPolynomial* mBP;
96  MatrixMxN mK; // K[i][j] = coefficient for Bi(x)*Bj(y)
97 };
